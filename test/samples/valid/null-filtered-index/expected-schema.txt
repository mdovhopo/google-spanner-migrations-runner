CREATE TABLE migrations_log (
  id STRING(64) NOT NULL,
  success BOOL NOT NULL,
  appliedAt TIMESTAMP NOT NULL,
  error STRING(4096),
) PRIMARY KEY(id)
CREATE TABLE data (
  id STRING(36) NOT NULL,
  deduplication_key STRING(MAX),
) PRIMARY KEY(id)
CREATE UNIQUE NULL_FILTERED INDEX unique_active_task ON data(id, deduplication_key)
